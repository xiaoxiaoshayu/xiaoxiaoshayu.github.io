import turtle
import random

turtle.shape("turtle")
max_radius=200
turtle.color("orange","green")
turtle.pensize(5)

turtle.penup()
turtle.right(90)
turtle.forward(max_radius)
turtle.left(90)
turtle.pendown()

turtle.speed(5)
turtle.circle(max_radius)
turtle.penup()
turtle.goto(0,0)
turtle.speed(0)


angle=0
distance=0
step=1


max_distance=0
while distance<=max_radius:
    if turtle.distance(0,0)>=max_radius-1:
        turtle.goto(0,0)
        angle=turtle.towards(0,0)+random.randint(-180,180)

        turtle.setheading(angle)
        distance=0

    turtle.forward(step)
    distance+=step
    max_distance+=step
    if max_distance>=max_radius*10:
        
        max_distance=0
        continue
